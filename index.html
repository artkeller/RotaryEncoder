RotaryEncoder

Rotary Encoder Library for measuring quadrature encoded signals

Rotary Encoder debounced - MCU: ARDUINO Family

Synopsis

#include "RotaryEncoder.h"

// -- Default initializing of Rotary Encoder
//    CLK = pin 3, DAT = pin 2, SW = pin A1, CW = count up, CCW = count down
//    One instance only is allowed due to plain ISR usage with global data
RotaryEncoder rotaryEncoder = RotaryEncoder();  

// -- Aternative initialization
//    Individual initializing of Rotary Encoder
//    CLK = pin 2, DAT = pin 3, SW = pin 4, CW = count down, CCW = count up`
RotaryEncoder rotaryEncoder = RotaryEncoder(2,3,4);     

// -- Apply Rotary Encoder methods
void loop() {  
  ...
  // -- Has to be inside loop() to enable rotary debouncer
  rotaryEncoder.startDebouncer();
  // -- Enable Button to clear recent encoder position if pushed, set to 0
  rotaryEncoder.updateClearButton(); 

  // -- Verify if encoder position has changed
  if (rotaryEncoder.hasChanged()) { 
  // -- Update counter for the dial (values 0..15)
    volatile byte encoderPos = rotaryEncoder.encodedPos() % 16; 
    do_something_with(encoderPos);
  }
  ...
}
